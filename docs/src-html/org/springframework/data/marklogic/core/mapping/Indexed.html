001package org.springframework.data.marklogic.core.mapping;
002
003import java.lang.annotation.ElementType;
004import java.lang.annotation.Retention;
005import java.lang.annotation.RetentionPolicy;
006import java.lang.annotation.Target;
007
008@Target({ElementType.ANNOTATION_TYPE, ElementType.FIELD})
009@Retention(RetentionPolicy.RUNTIME)
010public @interface Indexed {
011    IndexType type() default IndexType.PATH;
012
013    /**
014     * Path within the JSON/XML document to get to the property.  For example, in the following structure:
015     *
016     * {
017     *     "name": "Bob",
018     *     "pets": [
019     *         { "name": "Fluffy" }
020     *     ]
021     * }
022     *
023     * the "path" would be set to "/pets" for the property "name" inside the pets array.  This is used as a hint to the
024     * database sorting if you are using full path range indexes to properties.
025     *
026     * @return
027     */
028   String path() default "";
029
030    // TODO: Info to drive automatically creating indexes?
031}
