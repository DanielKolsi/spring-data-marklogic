001package org.springframework.data.marklogic.core.mapping;
002
003import com.marklogic.client.io.Format;
004
005// TODO: Is it good to abstract away the MarkLogic-specific format for "supported" formats, or should we just use that since this library is MarkLogic-specific anyway?
006public enum DocumentFormat {
007    JSON,
008    XML;
009
010    public static DocumentFormat parse(Format format) {
011        switch (format) {
012            case XML: return XML;
013            case UNKNOWN:
014            case BINARY:
015            case TEXT:
016                throw new IllegalArgumentException("No supported document format for '" + format + "'");
017            default: return JSON;
018        }
019    }
020
021    public Format toFormat() {
022        switch (this) {
023            case XML: return Format.XML;
024            default: return Format.JSON;
025        }
026    }
027
028    @Override
029    public String toString() {
030        return super.toString().toLowerCase();
031    }
032}
