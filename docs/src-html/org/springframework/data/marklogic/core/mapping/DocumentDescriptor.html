001package org.springframework.data.marklogic.core.mapping;
002
003import com.marklogic.client.document.DocumentRecord;
004import com.marklogic.client.impl.JerseyServices;
005import com.marklogic.client.io.DocumentMetadataHandle;
006import com.marklogic.client.io.Format;
007import com.marklogic.client.io.marker.ContentHandle;
008
009public class DocumentDescriptor {
010
011    private String uri;
012    private DocumentMetadataHandle metadata;
013    private Format format;
014
015    // TODO - is there a "generic" way to do this for read/write?
016    private ContentHandle content;
017    private DocumentRecord record;
018
019    public DocumentDescriptor() {}
020
021    public DocumentDescriptor(String uri) {
022        this.uri = uri;
023    }
024
025    public DocumentDescriptor(String uri, ContentHandle content) {
026        this.uri = uri;
027        this.content = content;
028    }
029
030    public DocumentDescriptor(DocumentRecord record) {
031        this.uri = record.getUri();
032        this.record = record;
033    }
034
035    public String getUri() {
036        return uri;
037    }
038
039    public void setUri(String uri) {
040        this.uri = uri;
041    }
042
043    public ContentHandle getContent() {
044        return content;
045    }
046
047    public void setContent(ContentHandle content) {
048        this.content = content;
049    }
050
051    public DocumentRecord getRecord() {
052        return record;
053    }
054
055    public void setRecord(DocumentRecord record) {
056        this.record = record;
057    }
058
059    public DocumentMetadataHandle getMetadata() {
060        return metadata;
061    }
062
063    public void setMetadata(DocumentMetadataHandle metadata) {
064        this.metadata = metadata;
065    }
066
067    public Format getFormat() {
068        return format;
069    }
070
071    public void setFormat(Format format) {
072        this.format = format;
073    }
074}
