001package org.springframework.data.marklogic.core.mapping;
002
003import com.marklogic.client.io.Format;
004import org.springframework.beans.BeansException;
005import org.springframework.context.ApplicationContext;
006import org.springframework.context.ApplicationContextAware;
007import org.springframework.data.mapping.model.BasicPersistentEntity;
008import org.springframework.data.util.TypeInformation;
009
010import java.util.Comparator;
011
012import static org.springframework.data.marklogic.core.Util.coalesce;
013
014public class BasicMarkLogicPersistentEntity<T> extends BasicPersistentEntity<T, MarkLogicPersistentProperty> implements
015        MarkLogicPersistentEntity<T>, ApplicationContextAware {
016
017    private TypePersistenceStrategy typePersistenceStrategy;
018    private Format documentFormat;
019    private String baseUri;
020    private String typeName;
021    private String dbSerializer;
022    private String dbDeserializer;
023    private Format dbFormat;
024
025    public BasicMarkLogicPersistentEntity(TypeInformation<T> information) {
026        this(information, null);
027    }
028
029    public BasicMarkLogicPersistentEntity(TypeInformation<T> information, Comparator<MarkLogicPersistentProperty> comparator) {
030        super(information, comparator);
031
032        Document document = this.findAnnotation(Document.class);
033        TypePersistenceStrategy defaultTypeStrategy = TypePersistenceStrategy.COLLECTION;
034        String defaultTypeName = information.getType().getSimpleName();
035        Format defaultFormat = Format.JSON;
036        String defaultUri = normalize(defaultTypeName);
037        String defaultDbSerializer = null;
038        String defautlDbDeserializer = null;
039        Format defaultDbFormat = Format.JSON;
040
041        if (document != null) {
042            this.baseUri = normalize(coalesce(document.uri(), document.value(), defaultUri));
043            this.documentFormat = document.format().toFormat();
044            this.typePersistenceStrategy = document.typeStrategy();
045            // TODO: if configuration says use full name instead of simple name, let that be the default
046            this.typeName = coalesce(document.type(), defaultTypeName);
047            this.dbSerializer = document.dbSerializer();
048            this.dbDeserializer = document.dbDeserializer();
049            this.dbFormat = document.dbFormat().toFormat();
050        } else {
051            this.baseUri = defaultUri;
052            this.typePersistenceStrategy = defaultTypeStrategy;
053            this.documentFormat = defaultFormat;
054            this.typeName = defaultTypeName;
055            this.dbSerializer = defaultDbSerializer;
056            this.dbDeserializer = defautlDbDeserializer;
057            this.dbFormat = defaultDbFormat;
058        }
059    }
060
061    @Override
062    public String getBaseUri() {
063        return baseUri;
064    }
065
066    @Override
067    public String getCollection() {
068        return null;
069    }
070
071    public TypePersistenceStrategy getTypePersistenceStrategy() {
072        return this.typePersistenceStrategy;
073    }
074
075    @Override
076    public Format getDocumentFormat() {
077        return this.documentFormat;
078    }
079
080    @Override
081    public void setApplicationContext(ApplicationContext applicationContext) throws BeansException {
082
083    }
084
085    @Override
086    public String getTypeName() {
087        return typeName;
088    }
089
090    @Override
091    public String getDbSerializer() {
092        return dbSerializer;
093    }
094
095    @Override
096    public String getDbDeserializer() {
097        return dbDeserializer;
098    }
099
100    @Override
101    public Format getDbFormat() {
102        return dbFormat;
103    }
104
105    private String normalize(String uri) {
106        String result = uri;
107        if (!result.startsWith("/")) {
108            result = "/" + result;
109        }
110        if (!result.endsWith("/")) {
111            result = result + "/";
112        }
113        return result;
114    }
115}
