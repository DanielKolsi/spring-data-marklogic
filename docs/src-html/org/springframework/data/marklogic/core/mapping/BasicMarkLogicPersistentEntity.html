001package org.springframework.data.marklogic.core.mapping;
002
003import com.marklogic.client.io.Format;
004import org.springframework.beans.BeansException;
005import org.springframework.context.ApplicationContext;
006import org.springframework.context.ApplicationContextAware;
007import org.springframework.data.mapping.model.BasicPersistentEntity;
008import org.springframework.data.marklogic.core.convert.ServerTransformer;
009import org.springframework.data.util.TypeInformation;
010
011import java.util.Comparator;
012
013import static org.springframework.data.marklogic.core.Util.coalesce;
014
015public class BasicMarkLogicPersistentEntity<T> extends BasicPersistentEntity<T, MarkLogicPersistentProperty> implements
016        MarkLogicPersistentEntity<T>, ApplicationContextAware {
017
018    private TypePersistenceStrategy typePersistenceStrategy;
019    private Format documentFormat;
020    private String baseUri;
021    private String typeName;
022    private Class<? extends ServerTransformer> transformer;
023
024    public BasicMarkLogicPersistentEntity(TypeInformation<T> information) {
025        this(information, null);
026    }
027
028    public BasicMarkLogicPersistentEntity(TypeInformation<T> information, Comparator<MarkLogicPersistentProperty> comparator) {
029        super(information, comparator);
030
031        Document document = this.findAnnotation(Document.class);
032        TypePersistenceStrategy defaultTypeStrategy = TypePersistenceStrategy.COLLECTION;
033        String defaultTypeName = information.getType().getSimpleName();
034        Format defaultFormat = Format.JSON;
035        String defaultUri = normalize(defaultTypeName);
036        Class<? extends ServerTransformer> defaultTransformer = null;
037
038        if (document != null) {
039            this.baseUri = normalize(coalesce(document.uri(), document.value(), defaultUri));
040            this.documentFormat = document.format().toFormat();
041            this.typePersistenceStrategy = document.typeStrategy();
042            // TODO: if configuration says use full name instead of simple name, let that be the default
043            this.typeName = coalesce(document.type(), defaultTypeName);
044            this.transformer = document.transformer();
045        } else {
046            this.baseUri = defaultUri;
047            this.typePersistenceStrategy = defaultTypeStrategy;
048            this.documentFormat = defaultFormat;
049            this.typeName = defaultTypeName;
050            this.transformer = defaultTransformer;
051        }
052    }
053
054    @Override
055    public String getBaseUri() {
056        return baseUri;
057    }
058
059    @Override
060    public String getCollection() {
061        return null;
062    }
063
064    public TypePersistenceStrategy getTypePersistenceStrategy() {
065        return this.typePersistenceStrategy;
066    }
067
068    @Override
069    public Format getDocumentFormat() {
070        return this.documentFormat;
071    }
072
073    @Override
074    public void setApplicationContext(ApplicationContext applicationContext) throws BeansException {
075
076    }
077
078    @Override
079    public String getTypeName() {
080        return typeName;
081    }
082
083    @Override
084    public Class<? extends ServerTransformer> getTransformer() {
085        return transformer;
086    }
087
088    private String normalize(String uri) {
089        String result = uri;
090        if (!result.startsWith("/")) {
091            result = "/" + result;
092        }
093        if (!result.endsWith("/")) {
094            result = result + "/";
095        }
096        return result;
097    }
098}
