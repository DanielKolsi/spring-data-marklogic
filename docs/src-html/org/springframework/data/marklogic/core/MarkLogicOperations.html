022/**
023 * The central interface between you and the MarkLogic database.  Helper methods have been created for basic CRUD
024 * operations as well as for building some common query constructs.  Additional methods allow you direct access to the
025 * different query and document managers, as well as direct access to the database client, so that you can write logic
026 * against the database using the full power of the MarkLogic Java Client Library.
027 *
028 * The main implementation of this is {@link MarkLogicTemplate}, so you will create an instance of that
029 * in order to use this interface.
030 */
031public interface MarkLogicOperations {
032
033    // ========== Query Building =========== //
034
035    /**
036     * Get an instance of a {@link com.marklogic.client.pojo.PojoQueryBuilder} for the specified entity class.  This is a simplified interface for
037     * building structured queries against JSON objects in the database.  For full control of all query options use
038     * the {@link com.marklogic.client.query.StructuredQueryBuilder} instead.
039     *
040     * @param entityClass Type of Java entity being stored in the database.
041     * @param <T> The type of the entity.
042     * @return
043     */
044    <T> PojoQueryBuilder<T> qb(Class<T> entityClass);
045
046    /**
047     * Adds sorting configuration to the specified query.
048     *
049     * @see MarkLogicOperations#sortQuery(Sort, StructuredQueryDefinition, Class)
050     */
051    StructuredQueryDefinition sortQuery(Sort sort, StructuredQueryDefinition query);
052
053    /**
054     * Add sorting configuration to the specified query.  The default sort algorithm will expect to use a path range
055     * index with specified sort properties, i.e. if sorting on "name" then a path index of "/name" should exist.
056     *
057     * Through use of the {@link org.springframework.data.marklogic.core.mapping.Indexed} annotation you can indicate
058     * use of a different type of range index for the property sorting, or specify the full path that should be used in
059     * creation of the sort options.  This requires that the entity type is specified.
060     *
061     * @param sort Sort information for the query, i.e. which properties and which orders.
062     * @param query The structured query to "enhance".
063     * @param entityClass The entity type.
064     *
065     * @return The "enhanced" query definition for use in continued building.
066     */
067    <T> StructuredQueryDefinition sortQuery(Sort sort, StructuredQueryDefinition query, Class<T> entityClass);
068
069    /**
070     * Add a search term to the specified query.  This is used for word searches across your entire document.  For more
071     * fine-grained control use {@link com.marklogic.client.query.StructuredQueryBuilder#word(StructuredQueryBuilder.TextIndex, String...)}
072     * to specify specific properties or fields into which to scope the word search.
073     *
074     * @param term A search phrase.
075     * @param query The structured query to "enhance".
076     *
077     * @return The "enhanced" query definition for use in continued building.
078     */
079    StructuredQueryDefinition termQuery(String term, StructuredQueryDefinition query);
080
081    // ========== Database Configuration =========== //
082
083    /**
084     * Configures the connected database using the specified JSON information.  The format of the JSON is outlined at
085     * http://docs.marklogic.com/REST/PUT/manage/v2/databases/[id-or-name]/properties.  This requires that the client
086     * was created with a user that has "admin" privileges.
087     *
088     * @param configuration Resource pointing to a JSON document containing database configuration information.
089     * @throws IOException
090     */
091    void configure(Resource configuration) throws IOException;
092
093    // ========== Database Operation Execution =========== //
094
095    /**
096     * Executes the specified action using a {@link com.marklogic.client.document.GenericDocumentManager} interface. For
097     * all the basic operations this is sufficient.
098     *
099     * @param action A function to execute using the document manager.  The active transaction is also provided so that
100     *               the function can add additional operations to it.
101     * @param <T> The result type of the function.
102     *
103     * @return The results of the document manager interactions (results of a query, etc.).
104     */
105    <T> T execute(DocumentCallback<T> action);
106
107    /**
108     * Executes the specified action using the full {@link com.marklogic.client.DatabaseClient}.  This gives you full
109     * control of your interactions with the database with the full power of the MarkLogic Java Client Library, but it
110     * also requires you to do a lot of the busy work yourself.  Should mainly be used for operations not supported
111     * currently in this template interface either through the helper methods, a {@link com.marklogic.client.document.GenericDocumentManager},
112     * or a {@link com.marklogic.client.query.QueryManager}.
113     *
114     * @param action A function to execute using the database client and active transaction.
115     * @param <T> The result type of the function.
116     *
117     * @return The results of the function.
118     */
119    <T> T executeWithClient(ClientCallback<T> action);
120
121    /**
122     * Executes the specified action using a {@link com.marklogic.client.query.QueryManager}.
123     *
124     * @param action A function to execute using the query manager and active transaction.
125     * @param <T> The result type of the function.
126     *
127     * @return The results from the query manager.
128     */
129    <T> T executeQuery(QueryCallback<T> action);

132
133    /**
134     * Write a single entity to the database into the default entity collection without any transforms.
135     *
136     * @see MarkLogicOperations#write(List, ServerTransform, String...)
137     */
138    <T> T write(T entity);
139
140    /**
141     * Write a single entity to the database into the specified collections, without any transforms.
142     *
143     * @see MarkLogicOperations#write(List, ServerTransform, String...)
144     */
145    <T> T write(T entity, String... collections);
146
147    /**
148     * Write the specified entity to the database into the default entity collection using the specified transform.
149     *
150     * @see MarkLogicOperations#write(List, ServerTransform, String...)
151     */
152    <T> T write(T entity, ServerTransform transform);
153
154    /**
155     * Write the specified entity to the database into the specified collections with the specified transform.
156     *
157     * @see MarkLogicOperations#write(List, ServerTransform, String...)
158     */
159    <T> T write(T entity, ServerTransform transform, String... collections);
160
161    /**
162     * Write the specified entities to the database into the default entity collection, without any transforms
163     *
164     * @see MarkLogicOperations#write(List, ServerTransform, String...)
165     */
166    <T> List<T> write(List<T> entities);
167
168    /**
169     * Write the specified entities to the database into the specified collections, without any transforms.
170     *
171     * @see MarkLogicOperations#write(List, ServerTransform, String...)
172     */
173    <T> List<T> write(List<T> entities, String... collections);
174
175    /**
176     * Write the specified entities to the database into the default entity collection using the specified transform.
177     *
178     * @see MarkLogicOperations#write(List, ServerTransform, String...)
179     */
180    <T> List<T> write(List<T> entities, ServerTransform transform);
181
182    /**
183     * Write the specified list of entities to the database.  Using the {@link org.springframework.data.marklogic.core.mapping.Document}
184     * annotation you can specify some information as to how it is persisted (i.e. whether or not to store it in a "type"
185     * collection, what path, etc.).
186     *
187     * If a transform is specified then before the entity is saved into the database they transform will be run against
188     * each entity.  For more information see http://docs.marklogic.com/guide/java/transforms.
189     *
190     * @param entities A list of POJO entities you wish to save into the database.
191     * @param transform The transform to use before finally persisting to the database.
192     * @param collections Additional collections to which you want the document to be a part of inside the database.
193     * @param <T> The type of entity
194     *
195     * @return The list of entities that were saved to the database.  This is mainly for convenience as they are not
196     * modified through this process.
197     */
198    <T> List<T> write(List<T> entities, ServerTransform transform, String... collections);
199
200    // ========== Database Reads =========== //
201
202    /**
203     * Reads a list of documents from the database by their URIs instead of ids.  Since type information is not know a
204     * "raw" {@link com.marklogic.client.document.DocumentRecord} is returned, and can be handled as desired.
205     *
206     * @param uris A list of database URIs of documents.
207     *
208     * @return A list of {@link com.marklogic.client.document.DocumentRecord} objects from the database.
209     */
210    List<DocumentRecord> read(List<?> uris);
211
212    /**
213     * Read a document with the specified ID from the database.
214     *
215     * @see MarkLogicOperations#read(List, Class)
216     */
217    <T> T read(Object id, Class<T> entityClass);
218
219    /**
220     * Reads a set of documents that have the specified IDs.  The documents must match the specified type or they will
221     * not be included in the result set.
222     *
223     * @param ids A list of IDs of entities.
224     * @param entityClass The Java type of the entity.
225     * @param <T> The type of the entity.
226     *
227     * @return A list of all the entities that matched the specified IDs.
228     */
229    <T> List<T> read(List<?> ids, Class<T> entityClass);

231    // ========== Database Queries without Entity  =========== //
232
233    /**
234     * Query for a list of documents using the specified structured query.
235     *
236     * @see MarkLogicOperations#search(StructuredQueryDefinition, int, int)
237     */
238    List<DocumentRecord> search(StructuredQueryDefinition query);
239
240    /**
241     * Query for a page of documents using the specified structured query, starting at the specified item in the
242     * result set.
243     *
244     * @see MarkLogicOperations#search(StructuredQueryDefinition, int, int)
245     */
246    DocumentPage search(StructuredQueryDefinition query, int start);
247
248    /**
249     * Query for a page of documents using the specified structured query.  Specify a start and length to get different
250     * "chunks" of data in the result set from the database.  You seldom want to return all results, as this could be a
251     * very large number and will degrade the performance of your application.  You instead should query out "pages" of
252     * data as needed.
253     *
254     * @param query A structured query defining the constraints to use to match the desired documents.
255     * @param start The start index within the result set that acts as the start of the "page".
256     * @param length The number of items to include in the "page"
257     *
258     * @return A {@link com.marklogic.client.document.DocumentPage} containing the records for this "chunk" of the result
259     * set.
260     */
261    DocumentPage search(StructuredQueryDefinition query, int start, int length);
262
263    /**
264     * Same as search with int bounds, but allows paging/sorting based off Spring Pageable.
265     *
266     * @see MarkLogicOperations#search(StructuredQueryDefinition, int, int, Class)
267     */
268    DocumentPage search(StructuredQueryDefinition query, Pageable pageable);
269
270    // ========== Database Queries with Entity =========== //
271
272    /**
273     * Query for a single entity using the specified structured query.  If for some reason multiple results are matched
274     * then only the first is returned.
275     *
276     * This is a convenience method to get a single item when you "know" that is all there is.
277     *
278     * @see MarkLogicOperations#search(StructuredQueryDefinition, int, int, Class)
279     */
280    <T> T searchOne(StructuredQueryDefinition query, Class<T> entityClass);
281
282    /**
283     * Query for a list of of documents of the specified type using the specified structured query.  Only the first
284     * {@link com.marklogic.client.impl.DocumentManagerImpl#DEFAULT_PAGE_LENGTH} documents are returned.
285     *
286     * @see MarkLogicOperations#search(StructuredQueryDefinition, int, int, Class)
287     */
288    <T> List<T> search(StructuredQueryDefinition query, Class<T> entityClass);
289
290    /**
291     * Query for a page of of documents of the specified type using the specified structured query.  Only a page of
292     * documents is returned, starting from the specified start index and containing
293     * {@link com.marklogic.client.impl.DocumentManagerImpl#DEFAULT_PAGE_LENGTH} entities.
294     *
295     * @see MarkLogicOperations#search(StructuredQueryDefinition, int, int, Class)
296     */
297    <T> Page<T> search(StructuredQueryDefinition query, int start, Class<T> entityClass);
298
299    /**
300     * Queries for a page of documents of the specified type, using the specified structured query to constrain which
301     * documents are returned.  Only a page is returned, starting from the specified index and containing the specified
302     * number of entries.
303     *
304     * @param query The structured query to use to match documents in the database.
305     * @param start The starting index within the result set of matches.
306     * @param limit The number of documents to return.
307     * @param entityClass The entity type class.
308     * @param <T> The type of entity.
309     *
310     * @return A page of documents matching the specified parameters.
311     */
312    <T> Page<T> search(StructuredQueryDefinition query, int start, int limit, Class<T> entityClass);
313
314    /**
315     * Same as search with int bounds, but allows paging/sorting based off Spring Pageable.
316     *
317     * @see MarkLogicOperations#search(StructuredQueryDefinition, int, int, Class)
318     */
319    <T> Page<T> search(StructuredQueryDefinition query, Pageable pageable, Class<T> entityClass);
320
321    /**
322     * @see MarkLogicOperations#facetedSearch(StructuredQueryDefinition, int, int, Class)
323     */
324    <T> FacetedPage<T> facetedSearch(StructuredQueryDefinition query, int start, Class<T> entityClass);
325
326    /**
327     * Similar to a normal paged query, except also includes facets for the matched results.  In order for the facets to
328     * be generated you need to ensure the specified query has the name of persisted options that describe the facets,
329     * or the query is a {@link org.springframework.data.marklogic.repository.query.CombinedQueryDefinition} that includes
330     * the ad-hoc options definitions for them.
331     *
332     * @param query The structured query to use to match documents in the database.
333     * @param start The starting index within the result set of matches.
334     * @param limit The number of documents to return.
335     * @param entityClass The entity type class.
336     * @param <T> The type of entity.
337     *
338     * @return A page of documents matching the specified parameters with the included facets.
339     */
340    <T> FacetedPage<T> facetedSearch(StructuredQueryDefinition query, int start, int limit, Class<T> entityClass);
341
342    /**
343     * Same as faceted search with int bounds, but allows paging/sorting based off Spring Pageable.
344     *
345     * @see MarkLogicOperations#search(StructuredQueryDefinition, int, int, Class)
346     */
347    <T> FacetedPage<T> facetedSearch(StructuredQueryDefinition query, Pageable pageable, Class<T> entityClass);

377     * Same as method taking int bounds, but allows use of Pagable for paging/sorting.
378     *
379     * @see MarkLogicOperations#stream(StructuredQueryDefinition, int, int, Class)

381    InputStream stream(StructuredQueryDefinition query, Pageable pageable);
382
383    /**
384     * Queries just like the {@link MarkLogicOperations#search(StructuredQueryDefinition, int, int, Class)} method does,
385     * but instead of a page/list of documents it returns an input stream straight from the results of the REST call
386     * against the database.  The purpose of these methods is to skip the serialization/deserialization process of
387     * your application layer and just return raw data.
388     *
389     * If you don't need to access any parts of the document and just need to return the raw
390     * JSON/XML then it is easier/faster to just copy the stream from the database into the HTTP response directly.
391     *
392     * Extractions or server-side transforms can still be used to change how the data is returned in cases where there
393     * are properties you don't want returned through a public API, etc.
394     *
395     * @return An input stream of all the documents that matched the specified parameters.
396     */
397    <T> InputStream stream(StructuredQueryDefinition query, int start, int length, Class<T> entityClass);
398
399    /**
400     * Same as method taking int bounds, but allows use of Pagable for paging/sorting.
401     *
402     * @see MarkLogicOperations#stream(StructuredQueryDefinition, int, int, Class)
403     */
404    <T> InputStream stream(StructuredQueryDefinition query, Pageable pageable, Class<T> entityClass);

406    // ========== Database Existence Checks =========== //
407
408    /**
409     * Check for existence of a document by using the full URI under which the document is stored.
410     *
411     * @param uri The full URI of the document, i.e. "/Person/my-id.json"

413     * @return True if a document with the specified URI exists.
414     */

417    /**

428    /**

552    // ========== Utility Methods =========== //

576    QueryConversionService getQueryConversionService();
