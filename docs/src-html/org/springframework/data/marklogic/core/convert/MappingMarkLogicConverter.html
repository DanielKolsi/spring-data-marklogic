041public class MappingMarkLogicConverter implements MarkLogicConverter, InitializingBean {
042
043    private static final Logger LOG = LoggerFactory.getLogger(MappingMarkLogicConverter.class);
044
045    private static final ConversionService converter = new DefaultConversionService();
046
047    private static final Pattern formats = Pattern.compile("\\.(json|xml)");
048
049    public static final String ISO_8601_FORMAT = "yyyy-MM-dd'T'HH:mm:ss.SSSXXX";
050    public static SimpleDateFormat simpleDateFormat8601 = new SimpleDateFormat(ISO_8601_FORMAT);
051    static { simpleDateFormat8601.setTimeZone(TimeZone.getTimeZone("UTC")); }
052
053    private MappingContext<? extends MarkLogicPersistentEntity<?>, MarkLogicPersistentProperty> mappingContext;
054    private ObjectMapper objectMapper;
055    private ObjectMapper xmlMapper;
056
057    public MappingMarkLogicConverter(MappingContext<? extends MarkLogicPersistentEntity<?>, MarkLogicPersistentProperty> mappingContext) {
058        this.mappingContext = mappingContext;
059    }
060
061    @Override
062    public MappingContext<? extends MarkLogicPersistentEntity<?>, MarkLogicPersistentProperty> getMappingContext() {
063        return mappingContext;
064    }
065
066    @Override
067    public ConversionService getConversionService() {
068        return null;
069    }

071    @Override
072    public void write(Object source, DocumentDescriptor doc) {
073        final MarkLogicPersistentEntity<?> entity = getMappingContext().getPersistentEntity(source.getClass());
074
075        if (entity != null && entity.hasIdProperty()) {
076            PersistentProperty idProperty = entity.getPersistentProperty(entity.getIdProperty().getName());
077
078            if (Collection.class.isAssignableFrom(idProperty.getType()) || Map.class.isAssignableFrom(idProperty.getType()))
079                throw new IllegalArgumentException("Collection types not supported as entity id");
080
081            try {
082                // TODO: Better way to do this?
083                // TODO: Support document URI templates if the ID value is null
084                Object id = idProperty.getGetter().invoke(source);
085                doc.setUri(getDocumentUris(asList(id), entity.getType()).get(0));
086            } catch (Exception e) {
087                throw new IllegalArgumentException("Unable to access value of @Id from " + idProperty.getName());
088            }
089        } else {
090            throw new IllegalArgumentException("Your entity of type " + source.getClass().getName() + " does not have a method or field annotated with org.springframework.data.annotation.Id");
091        }
092
093        if (entity.getTypePersistenceStrategy() == TypePersistenceStrategy.COLLECTION) {
094            if (doc.getMetadata() == null) doc.setMetadata(new DocumentMetadataHandle());
095            doc.setMetadata(doc.getMetadata().withCollections(entity.getTypeName()));
096        } else {
097            doc.setMetadata(new DocumentMetadataHandle());
098        }
099
100        JacksonDatabindHandle contentHandle = new JacksonDatabindHandle<>(source);
101        if (mapAsXml(entity)) {
102            contentHandle.setMapper(xmlMapper);
103        } else {
104            contentHandle.setMapper(objectMapper);
105        }
106
107        doc.setFormat(entity.getDocumentFormat());
108        doc.setContent(contentHandle);
109    }
110
111    @Override
112    public <R> R read(Class<R> clazz, DocumentDescriptor doc) {
113        final MarkLogicPersistentEntity<?> entity = getMappingContext().getPersistentEntity(clazz);
114
115        JacksonDatabindHandle<R> handle = new JacksonDatabindHandle<>(clazz);
116        if (mapAsXml(entity)) {
117            handle.setMapper(xmlMapper);
118        } else {
119            handle.setMapper(objectMapper);
120        }
121
122        R mapped = doc.getRecord().getContent(handle).get();
123
124        // We assume that the ID from the database is the correct one, so update the property with the @Id annotation with the "correct" ID
125        if (entity != null && entity.hasIdProperty() && mapped != null) {
126            PersistentProperty idProperty = entity.getPersistentProperty(entity.getIdProperty().getName());
127            try {
128                // TODO: Better way to do this?
129                Method setter = idProperty.getSetter();
130                if (setter != null) setter.invoke(mapped, uriToId(doc.getUri(), entity.getDocumentFormat(), entity.getBaseUri(), idProperty.getType()));
131            } catch (Exception e) {
132                throw new IllegalArgumentException("Unable to set value of @Id from " + idProperty.getName());
133            }
134        }
135
136        return mapped;
137    }
138
139    @Override
140    public List<String> getDocumentUris(List<?> ids) {
141        return getDocumentUris(ids, null);
142    }
143
144    @Override
145    public <T> List<String> getDocumentUris(List<?> ids, Class<T> entityClass) {
146
147        final List<String> uris = ids.stream()
148                .flatMap(id -> {
149                    if (entityClass != null) {
150                        final MarkLogicPersistentEntity<?> entity = getMappingContext().getPersistentEntity(entityClass);
151                        return Stream.of(idToUri(id, entity.getDocumentFormat(), entity.getBaseUri()));
152                    } else {
153                        // Just from the ID we don't know the type, or can't infer it, so we need to "try" both.  The potential downside
154                        // is if they have both JSON/XML for the same id - might get "odd" results?
155                        return Stream.of(
156                                idToUri(id, Format.JSON, "/"),
157                                idToUri(id, Format.XML, "/")
158                        );
159                    }
160                })
161                .collect(Collectors.toList());
162
163        return uris;
164    }
165
166    @Override
167    public boolean mapAsXml(MarkLogicPersistentEntity entity) {
168        return entity != null && entity.getDocumentFormat() == Format.XML && xmlMapper != null;
169    }
170
171    private Object uriToId(String uri, Format format, String baseUri, Class<?> idType) {
172        String id = formats
173                .matcher(uri.substring(baseUri.length()))
174                .replaceAll("");
175        return converter.convert(id, idType);
176    }
177
178    private String idToUri(Object id, Format format, String baseUri) {
179        return baseUri + String.valueOf(id) + "." + format.toString().toLowerCase();
180    }
181
182    @Override
183    public <T> QueryDefinition wrapQuery(StructuredQueryDefinition query, Class<T> entityClass) {
184        boolean isRaw = query instanceof CombinedQueryDefinition && ((CombinedQueryDefinition) query).isQbe();
185        CombinedQueryDefinition combined = combine(query).type(entityClass);
186        if (isRaw) {
187            return combined.getRawQbe();
188        } else {
189            return combined;
190        }
191    }
192
193    @Override
194    public <T> ServerTransform getSerializer(Class<T> entityClass) {
195        ServerTransform serializer = null;
196
197        if (entityClass != null) {
198            MarkLogicPersistentEntity entity = getMappingContext().getPersistentEntity(entityClass);
199
200            if (entity != null && hasText(entity.getDbSerializer())) {
201                serializer = new ServerTransform(entity.getDbSerializer());
202            }
203        }
204
205        return serializer;
206    }
207
208    @Override
209    public <T> ServerTransform getDeserializer(Class<T> entityClass) {
210        ServerTransform serializer = null;
211
212        if (entityClass != null) {
213            MarkLogicPersistentEntity entity = getMappingContext().getPersistentEntity(entityClass);
214
215            if (entity != null && hasText(entity.getDbDeserializer())) {
216                serializer = new ServerTransform(entity.getDbDeserializer());
217            }
218        }
219
220        return serializer;
221    }
222
223    @Override
224    public void afterPropertiesSet() {
225        objectMapper = new ObjectMapper()
226                .configure(JsonGenerator.Feature.AUTO_CLOSE_TARGET, false)
227                .configure(JsonParser.Feature.AUTO_CLOSE_SOURCE, false)
228                .configure(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS, false)
229                .setDateFormat(simpleDateFormat8601)
230                .registerModule(new JavaTimeModule())
231                // Since we don't configure to "wrap" in the class name we can't do "type scoped" path range indexes - could be a problem options larger data sets
232                .disableDefaultTyping();
233
234        try {
235            // TODO: Is it just easier/better to include the dumb library?  It will cause the default behavior to change for Spring Web stuff
236            Class mapperClass = Class.forName("com.fasterxml.jackson.dataformat.xml.XmlMapper", false, this.getClass().getClassLoader());
237            xmlMapper = ((ObjectMapper) mapperClass.newInstance())
238                    .configure(JsonGenerator.Feature.AUTO_CLOSE_JSON_CONTENT, false)
239                    .configure(JsonGenerator.Feature.AUTO_CLOSE_TARGET, false)
240                    .configure(JsonParser.Feature.AUTO_CLOSE_SOURCE, false)
241                    .configure(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS, false)
242                    .setDateFormat(MappingMarkLogicConverter.simpleDateFormat8601)
243                    .registerModule(new JavaTimeModule())
244                    .disableDefaultTyping();
245        } catch (ClassNotFoundException e) {
246            LOG.warn("com.fasterxml.jackson.dataformat:jackson-dataformat-xml needs to be included in order to use Java->XML conversion");
247        } catch (IllegalAccessException e) {
248            LOG.warn("Unable to instantiate XmlMapper instance in order to use Java->XML conversion");
249        } catch (InstantiationException e) {
250            LOG.warn("Unable to instantiate XmlMapper instance in order to use Java->XML conversion");
251        }
252
253    }
