001package org.springframework.data.marklogic.core.convert;
002
003import com.marklogic.client.document.ServerTransform;
004import com.marklogic.client.query.QueryDefinition;
005import com.marklogic.client.query.StructuredQueryDefinition;
006import org.springframework.data.domain.Example;
007import org.springframework.data.marklogic.core.mapping.MarkLogicPersistentEntity;
008import org.springframework.data.marklogic.repository.query.CombinedQueryDefinition;
009import org.springframework.util.Assert;
010
011import java.util.concurrent.ConcurrentHashMap;
012
013import static org.springframework.data.marklogic.repository.query.CombinedQueryDefinitionBuilder.combine;
014
015/**
016 * Helper class for last-minute tweaks to the query, or for last minute updates to read/write operations
017 */
018public class QueryMapper {
019
020    private final MarkLogicConverter converter;
021    private final ConcurrentHashMap<Class, ServerTransformer> transformers = new ConcurrentHashMap<>();
022
023    public QueryMapper(MarkLogicConverter converter) {
024        Assert.notNull(converter, "MarkLogicConverter must not be null!");
025        this.converter = converter;
026    }
027
028    public QueryDefinition getMappedQuery(StructuredQueryDefinition query, Class entityClass) {
029        boolean isRaw = query instanceof CombinedQueryDefinition && ((CombinedQueryDefinition) query).isQbe();
030        CombinedQueryDefinition combined = combine(query).type(entityClass);
031
032        // If no server transform is already set on the query then we can see if there is a entity-configured transform
033        if (entityClass != null && combined.getResponseTransform() == null) {
034            combined.setResponseTransform(getReadTransform(entityClass));
035        }
036
037        if (isRaw) {
038            return combined.getRawQbe();
039        } else {
040            return combined;
041        }
042    }
043
044    public <T> QueryDefinition getExampleQuery(Example<T> example) {
045        return null;
046    }
047
048    public ServerTransform getReadTransform(Class entityClass) {
049        MarkLogicPersistentEntity<?> entity = converter.getMappingContext().getPersistentEntity(entityClass);
050        ServerTransformer transformer = findTransformer(entity);
051        return transformer != null
052                ? transformer.reader(entity, null)
053                : null;
054    }
055
056    public ServerTransform getWriteTransform(Class entityClass) {
057        MarkLogicPersistentEntity<?> entity = converter.getMappingContext().getPersistentEntity(entityClass);
058        ServerTransformer transformer = findTransformer(entity);
059        return transformer != null
060                ? transformer.writer(entity)
061                : null;
062    }
063
064    public ServerTransformer getTransformer(Class<? extends ServerTransformer> transformerClass) {
065        ServerTransformer transformer = transformers.get(transformerClass);
066
067        if (transformer == null) {
068            try {
069                transformer = transformerClass.newInstance();
070                transformers.put(transformerClass, transformer);
071            } catch (Exception e) {
072                e.printStackTrace();
073            }
074
075        }
076
077        return transformer;
078    }
079
080    protected ServerTransformer findTransformer(MarkLogicPersistentEntity<?> pEntity) {
081        ServerTransformer transformer = transformers.get(pEntity);
082
083        if (transformer == null) {
084            if (pEntity.getTransformer() != null && pEntity.getTransformer() != ServerTransformer.class) {
085                transformer = getTransformer(pEntity.getTransformer());
086            }
087        }
088
089        return transformer;
090    }
091}
