001package org.springframework.data.marklogic.core.convert;
002
003import com.marklogic.client.document.ServerTransform;
004import org.springframework.data.marklogic.core.mapping.MarkLogicPersistentEntity;
005import org.springframework.data.repository.query.ParameterAccessor;
006
007/**
008 * Interface for class that creates a configured server transform for read and write operations.  The default functionality
009 * for server transforms is that there are none, so all the methods ultimately default to returning null.  Any of the
010 * methods needed may be implemented to change that.
011 *
012 * Implemented the methods that take the entity type and the params to make a default for all read or write operations.
013 *
014 * If your implementation class has a constructor with a param of type {@link MarkLogicConverter} then the instantiated
015 * converter instance will be injected in when your class is instantiated.  If you don't need the converter then
016 */
017public interface ServerTransformer {
018
019    /**
020     * Global read transform.  All template read operations use this transform, unless otherwise overridden in a
021     * function call in a custom repository implementation.
022     *
023     * When the repository query builder is creating MarkLogic queries from your repository functions it will pass in
024     * the parameters passed into the method.  You can perform any business logic and add the values to the
025     * {@link ServerTransform} object as well, if necessary.
026     *
027     * The entity configuration will always be passed - this can be used to inspect annotations or do other logic to get
028     * other information about the entity class
029     *
030     * @param pEntity The persistent entity configuration class which can be used to get information about the entity
031     * @param params Function parameters passed to a repository query function.  Perform business logic and merge them
032     *               into the {@link ServerTransform} parameters, if necessary.
033     */
034    default <T> ServerTransform reader(MarkLogicPersistentEntity<T> pEntity, ParameterAccessor params) {
035        return null;
036    }
037
038    /**
039     * Global write transform.  All template write operations use this transform, unless otherwise overridden in a
040     * function call in a custom repository implementation.
041     *
042     * @param pEntity The persistent entity configuration class which can be used to get information about the entity
043     */
044    default <T> ServerTransform writer(MarkLogicPersistentEntity<T> pEntity) {
045        return null;
046    }
047}
