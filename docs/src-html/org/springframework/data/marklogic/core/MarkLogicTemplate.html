101    /**
102     * Create a template interface using the specified database client.  Also provide a custom convert that handles the
103     * translation of entities between Java and MarkLogic.  You can also provide a custom query conversion service that will
104     * handle the conversion of different types used in finder queries and @Query queries.
105     *
106     * @param client The MarkLogic database client created from @see {@link DatabaseClientFactory}.
107     * @param converter A custom entity converter.
108     * @param queryConversionService A custom query object conversion service.
109     */
110    public MarkLogicTemplate(DatabaseClient client, MarkLogicConverter converter, QueryConversionService queryConversionService) {
111        Assert.notNull(client, "A database client object is required!");
112        this.client = client;
113        this.converter = converter == null ? getDefaultConverter() : converter;
114        this.queryConversionService = queryConversionService == null ? getDefaultQueryConverter() : queryConversionService;
115        this.exceptionTranslator = new MarkLogicExceptionTranslator();
116
117        // Create a RestTemplate instance for use directly against the REST API because there are some things that
118        // aren't fully supported in the client
119        HttpClient httpClient = HttpClientBuilder
120                .create()
121                .setDefaultCredentialsProvider(provider())
122                .build();
123        HttpComponentsClientHttpRequestFactory factory = new HttpComponentsClientHttpRequestFactory(httpClient);
124
125        this.restTemplate = new RestTemplate(factory);
126        services = ((DatabaseClientImpl) client).getServices();
127    }

129    private CredentialsProvider provider() {
130        // Get the username/password from the database client's security context
131        DatabaseClientFactory.SecurityContext securityContext = client.getSecurityContext();
132        String username;
133        String password;
134        if (securityContext instanceof DatabaseClientFactory.BasicAuthContext) {
135            username = ((DatabaseClientFactory.BasicAuthContext) securityContext).getUser();
136            password = ((DatabaseClientFactory.BasicAuthContext) securityContext).getPassword();
137        } else if (securityContext instanceof DatabaseClientFactory.DigestAuthContext) {
138            username = ((DatabaseClientFactory.DigestAuthContext) securityContext).getUser();
139            password = ((DatabaseClientFactory.DigestAuthContext) securityContext).getPassword();
140        } else {
141            throw new IllegalArgumentException("Currently only BasicAuthContext and DigestAuthContext are supported");
142        }
143
144        CredentialsProvider provider = new BasicCredentialsProvider();
145        UsernamePasswordCredentials credentials = new UsernamePasswordCredentials(username, password);
146        provider.setCredentials(new AuthScope(client.getHost(), AuthScope.ANY_PORT, AuthScope.ANY_REALM), credentials);
147        return provider;
148    }

291    @Override
292    public <T> List<T> write(List<T> entities, ServerTransform transform, String... collections) {
293        // TODO: Versioning?
294        List<DocumentDescriptor> docs =
295                entities.stream()
296                        .map(entity -> {
297                            final DocumentDescriptor doc = new DocumentDescriptor();
298                            this.converter.write(entity, doc);
299                            return doc;
300                        })
301                        .collect(Collectors.toList());
302
303        return execute((manager, transaction) -> {
304            if (!docs.isEmpty()) {
305                DocumentWriteSet writeSet = manager.newWriteSet();
306                for (DocumentDescriptor doc : docs) {
307                    if (collections != null && collections.length > 0) {
308                        // If collections are specified then those are the ones that will be used - we expect things to be how we specify
309                        doc.getMetadata().getCollections().clear();
310                        doc.setMetadata(doc.getMetadata().withCollections(collections));
311                    }
312
313                    if (doc.getUri() != null) {
314                        writeSet.add(doc.getUri(), doc.getMetadata(), doc.getContent());
315                    } else {
316                        DocumentUriTemplate template = manager.newDocumentUriTemplate(doc.getFormat().toString());
317                        writeSet.add((String) null, doc.getMetadata(), doc.getContent());
318                    }
319                }
320                manager.write(writeSet, transform, transaction);
321            }
322            return entities;
323        });
324    }

383     * Query for a list of of documents of the specified type.  As this returns all the documents for the type it can
384     * potentially take a long time to resolve pulling all of them back from the database.  The query will be fast, but
385     * transmitting all the records could time out if the result set is sufficiently large.  It is recommended that you
386     * page your results.
387     *
388     * This method is mainly created to accommodate creating an implementation for the
389     * {@link org.springframework.data.repository.CrudRepository#findAll()} method.
390     *
391     * @see MarkLogicOperations#search(StructuredQueryDefinition, int, int, Class)
392     */

394        return search(qb.and(), 0, -1, entityClass)
395                .getContent();

419    @Override
420    public <T> Page<T> search(StructuredQueryDefinition query, int start, int length, Class<T> entityClass) {
421        return execute((manager, transaction) -> {
422            if (length >= 0) manager.setPageLength(length);
423
424            QueryDefinition finalQuery = converter.wrapQuery(query, entityClass);
425            DocumentPage docPage = manager.search(finalQuery, start + 1, transaction);
426
427            List<T> results = toEntityList(entityClass, docPage);
428            return new PageImpl<>(results, new ChunkRequest(start, (int) manager.getPageLength()), docPage.getTotalSize());
429        });
430    }

457    @Override
458    public <T> InputStream stream(StructuredQueryDefinition query, int start, int length, Class<T> entityClass) {
459        return execute((manager, transaction) -> {
460            if (length >= 0) manager.setPageLength(length);
461
462            QueryDefinition finalQuery = converter.wrapQuery(query, entityClass);
463
464            DocumentPage page = manager.search(finalQuery, start + 1, transaction);
465
466            Enumeration<? extends InputStream> results = Collections.enumeration(
467                    StreamSupport.stream(page.spliterator(), true)
468                            .map(record -> record.getContentAs(InputStream.class))
469                            .collect(Collectors.toList()));
470
471            return new SequenceInputStream(results);
472        });
473    }

549    @Override
550    public <T> void delete(StructuredQueryDefinition query, Class<T> entityClass) {
551        executeWithClient((client, transaction) -> {
552            QueryManager qryMgr = client.newQueryManager();
553            CombinedQueryDefinition combined =
554                    combine(query).options("<values name='uris'><uri/></values>");
555
556            combined = (CombinedQueryDefinition) getConverter().wrapQuery(combined, entityClass);
557            ValuesDefinition valDef = qryMgr.newValuesDefinition("uris");
558            valDef.setQueryDefinition(
559                    qryMgr.newRawCombinedQueryDefinition(new StringHandle(combined.serialize()).withFormat(Format.XML))
560            );
561
562            ValuesHandle results = qryMgr.values(valDef, new ValuesHandle(), transaction);
563
564            // Convert the facets to just a list of the ids
565            List<String> uris = Arrays.stream(results.getValues())
566                    .map(value -> value.get("xs:string", String.class))
567                    .collect(Collectors.toList());
568
569            if (!uris.isEmpty()) {
570                client.newDocumentManager().delete(transaction, uris.toArray(new String[0]));
571            }
572
573            return null;
574        });
575    }

587    @Override
588    public void dropCollections(String... collections) {
589        executeQuery((manager, transaction) -> {
590            // The REST API only supports deleting one collection at a time, so we need to send a request for each
591            asList(collections).forEach(collection -> {
592                DeleteQueryDefinition deleteQuery = manager.newDeleteDefinition();
593                deleteQuery.setCollections(collection);
594                manager.delete(deleteQuery, transaction);
595            });
596            return null;
597        });
598    }

600    @Override
601    public <T> void dropCollection(Class<T> entityClass) {
602        // TODO: If no entity class is supplied do we just deleteById directory "/"?  Is that safe/good?
603        if (entityClass == null) {
604            throw new InvalidDataAccessApiUsageException("Entity class is required to determine scope of deleteById");
605        }
606
607        // If the type is not stored in a collection there is no "quick" way to mass deleteById other than directory.
608        // Can't easily do it by document properties
609        // TODO: Maybe uri lexicon for deleting without a collection, but can't assume it is on, though
610        MarkLogicPersistentEntity entity = converter.getMappingContext().getPersistentEntity(entityClass);
611        if (entity == null || entity.getTypePersistenceStrategy() != TypePersistenceStrategy.COLLECTION) {
612            throw new InvalidDataAccessApiUsageException(String.format("Cannot determine deleteById scope for entity of type %s", entityClass.getName()));
613        }
614
615        dropCollections(entityClass.getSimpleName());
616    }

618    @Override
619    public <T> void dropCollections(Class<T>... entityClasses) {
620        String[] collections = Arrays.stream(entityClasses)
621                .map(entityClass -> {
622                    MarkLogicPersistentEntity entity = converter.getMappingContext().getPersistentEntity(entityClass);
623                    if (entity == null || entity.getTypePersistenceStrategy() != TypePersistenceStrategy.COLLECTION) {
624                        throw new InvalidDataAccessApiUsageException(String.format("Cannot determine deleteById scope for entity of type %s", entityClass.getName()));
625                    }
626
627                    return entityClass.getSimpleName();
628                }).toArray(String[]::new);
629
630        dropCollections(collections);
631    }

642    private static RuntimeException potentiallyConvertRuntimeException(RuntimeException ex,
643                                                                       PersistenceExceptionTranslator exceptionTranslator) {
644        RuntimeException resolved = exceptionTranslator.translateExceptionIfPossible(ex);
645        return resolved == null ? ex : resolved;
646    }

648    private <T> List<T> toEntityList(Class<T> entityClass, DocumentPage page) {
649        final List<T> results = new ArrayList<>();
650        page.iterator().forEachRemaining(item -> {
651            results.add(converter.read(entityClass, new DocumentDescriptor(item)));
652        });
653        return results;
654    }
