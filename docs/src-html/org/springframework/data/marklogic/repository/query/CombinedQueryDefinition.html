001package org.springframework.data.marklogic.repository.query;
002
003import com.marklogic.client.document.ServerTransform;
004import com.marklogic.client.io.Format;
005import com.marklogic.client.query.RawQueryByExampleDefinition;
006import com.marklogic.client.query.StructuredQueryBuilder;
007import com.marklogic.client.query.StructuredQueryDefinition;
008import org.springframework.data.domain.Sort;
009import org.springframework.data.mapping.context.MappingContext;
010import org.springframework.data.marklogic.core.mapping.IndexType;
011
012import java.util.List;
013
014/**
015 * Convenience interface provided to allow easy creation of "combined" queries.  These type of queries combined structured
016 * queries and ad-hoc options (not persisted to the database) (see http://docs.marklogic.com/guide/rest-dev/search#id_69918)
017 * for more information.
018 *
019 * This interface follows the Java "builder" pattern to make it easier to construct queries.  As much as possible any
020 * defaults for these methods follow conventions used in the MarkLogic Java Client Library.
021 *
022 * This interface also has helper methods to add some of the more common query options like sorting, extractions, and
023 * term searches.
024 */
025public interface CombinedQueryDefinition extends StructuredQueryDefinition {
026
027    /**
028     * Gets the raw Query By Example definition from the combined query.
029     *
030     * @return the raw QBE object.
031     */
032    RawQueryByExampleDefinition getRawQbe();
033
034    /**
035     * Adds a raw Query By Example definition to the current combined query.
036     *
037     * @see CombinedQueryDefinitionBuilder#byExample(RawQueryByExampleDefinition, Format)
038     */
039    CombinedQueryDefinition byExample(RawQueryByExampleDefinition qbe);
040
041    /**
042     * Adds a raw Query By Example definition to the current combined query.  This will supercede any structured query
043     * that currently exists in the combined query, so use one or the other.
044     *
045     * @param qbe the raw QBE object.
046     * @param format Either JSON or XML, using the {@link com.marklogic.client.io.Format} enum.
047     *
048     * @return The current query definition for use in continued building.
049     */
050    CombinedQueryDefinition byExample(RawQueryByExampleDefinition qbe, Format format);
051
052    /**
053     * Combine the current structured queries in the combined query with the specified queries using an "and".
054     *
055     * @param queries Queries to add to the combined query
056     *
057     * @return The current query definition for use in continued building.
058     */
059    CombinedQueryDefinition and(StructuredQueryDefinition... queries);
060
061    /**
062     * Combine the current structured queries in the combined query with the specified queries using an "or".
063     *
064     * @param queries Queries to add to the combined query
065     *
066     * @return The current query definition for use in continued building.
067     */
068    CombinedQueryDefinition or(StructuredQueryDefinition... queries);
069
070    /**
071     * Adds constraints to limit results to only documents contained in one of the specified collections.  This can be
072     * called multiple times and the result is additive.
073     *
074     * By default all documents are in a collection named after the entity class, i.e. a Person entity is stored in a
075     * collection with the URI "Person".
076     *
077     * @param collections A list of collection URIs
078     *
079     * @return The current query definition for use in continued building.
080     */
081    CombinedQueryDefinition collections(String... collections);
082
083    /**
084     * Adds contraints to limit results to only documents contained in the specified directory.  Multiple calls to this
085     * function will replace the value of previous calls.
086     *
087     * By default all documents are stored under a path named after the entity class (to avoid ID clashes, since the
088     * database URI is the "true" primary key).  So a Person entity is stored under the path "/Person/".
089     *
090     * @param directory The directory path under which to limit results.
091     *
092     * @return The current query definition for use in continued building.
093     */
094    CombinedQueryDefinition directory(String directory);
095
096    /**
097     * Sets the query to use persisted options that have the specified name.  If you find you are adding lots of ad-hoc
098     * options to your queries or that you are having performance issues because of building them each time you can save
099     * those options and reference them in your query.
100     *
101     * For more information see http://docs.marklogic.com/guide/java/query-options#id_20346.
102     *
103     * @param name The name of the options as they are saved in the database.
104     *
105     * @return The current query definition for use in continued building.
106     */
107    CombinedQueryDefinition optionsName(String name);
108
109    /**
110     * Add a snippet of XML query options to the combined query.  If you wanted to configure the query to return the
111     * metrics of the query you would call the method like so:
112     *
113     *      combinedQuery.options("<return-metrics>true</return-metrics>");
114     *
115     * @param options Variable number of options XML strings to add to the query.
116     *
117     * @return The current query definition for use in continued building.
118     */
119    CombinedQueryDefinition options(String... options);
120
121    /**
122     * Add sorting configuration to the query.  The default sort algorithm will expect
123     * to use a path range index, i.e. if sorting on "name" then a path index of "/name" should exist.
124     *
125     * Through use of the {@link org.springframework.data.marklogic.core.mapping.Indexed} annotation you can indicate
126     * use of a different type of range index for the property sorting, or specify the full path that should be used in
127     * creation of the sort options.  This requires that {@link CombinedQueryDefinition#type(Class)} be called first to
128     * specify the entity type.
129     *
130     * @param sort Sort information for the query, i.e. which properties and which orders.
131     *
132     * @return The current query definition for use in continued building.
133     */
134    CombinedQueryDefinition sort(Sort sort);
135
136    /**
137     * Add sorting configuration to the query.  The type that is specified will be used on all sort properties specified
138     * in the {@link org.springframework.data.domain.Sort} object, and the assumption will be that how they are specified
139     * in that object is exactly how the index is defined in the configuration.  For example, if you have a sort property
140     * of "name" and you specify the type of IndexType.PATH then you would need a path range index defined for "name".  If the
141     * property is "/pets/name" with the same index type then a path range index would need to be defined for "/pets/name".
142     *
143     * Usually it is better to be consistent in how you define your indexes (i.e. use either path range or element (property)
144     * range indexes) so that the same type of index can be configured for many properties in a sort.
145     *
146     * @param sort Sort information for the query, i.e. which properties and which orders.
147     * @param type The type of index that is configured for the properties.  Used to create the correct options in the
148     *             combined query.
149     *
150     * @return The current query definition for use in continued building.
151     */
152    CombinedQueryDefinition sort(Sort sort, IndexType type);
153
154    /**
155     * Adds sorting configuration to the query.  This is the most fine-grained control of sort index configuration - you
156     * can specify details for each property.  The downside would be typically you would have to iterate over a
157     * {@link org.springframework.data.domain.Sort} yourself and perform logic to determine index type yourself.
158     *
159     * @param propertyName The name of the property, or path, or whatever (depending on index type used)
160     * @param order The order, either "descending" or "ascending"
161     * @param type The type of index that is configured for the properties.  Used to create the correct options in the
162     *             combined query.
163     *
164     * @return The current query definition for use in continued building.
165     */
166    CombinedQueryDefinition sort(String propertyName, String order, IndexType type);
167
168    /**
169     * Specify which properties of a document will be included/excluded from the results.
170     *
171     * @see CombinedQueryDefinition#extracts(List, SelectedMode)
172     */
173    CombinedQueryDefinition extracts(List<String> extracts);
174
175    /**
176     * Specify which properties of a document will be included/excluded from the results.  The default is for the entire
177     * document to be returned.
178     *
179     * Ideally you keep the mode as {@link SelectedMode#HIERARCHICAL} so that your entities will be de-serialized correctly.
180     * If you create a custom entity to handle the results of just the properties flattened out then you can just use
181     * the {@link SelectedMode#INCLUDE}.
182     *
183     * For more information see http://docs.marklogic.com/guide/java/searches#id_90087.
184     *
185     * @param extracts A list of XPaths that describe which properties in a document to include/exclude
186     * @param mode Specify whether to include or exclude the specified properties.
187
188     * @return The current query definition for use in continued building.
189     */
190    CombinedQueryDefinition extracts(List<String> extracts, SelectedMode mode);
191
192    /**
193     * Limit the results that come back to the specified number.  For configuring paging of the queries it is usually
194     * better to just specify your bounds in {@link org.springframework.data.marklogic.core.MarkLogicOperations#search(StructuredQueryDefinition, int, int, Class)}.
195     *
196     * @param limit Limit the results to this number of records.
197     *
198     * @return The current query definition for use in continued building.
199     */
200    CombinedQueryDefinition limit(int limit);
201
202    /**
203     * Specify a server transform to use on the results that are matched in the database.  This server transform will be
204     * applied to each document before the set is returned from the database, so for complicated transformation logic
205     * this could be more efficient than trying to do it in the Java layer.
206     *
207     * For more information see http://docs.marklogic.com/guide/java/transforms.
208     *
209     * @return The current query definition for use in continued building.
210     *
211     * @return
212     */
213    CombinedQueryDefinition transform(ServerTransform transform);
214
215    /**
216     * Add a search term to the current query.  This is used for word searches across your entire document.  For more
217     * fine-grained control use {@link com.marklogic.client.query.StructuredQueryBuilder#word(StructuredQueryBuilder.TextIndex, String...)}
218     * to specify specific properties or fields into which to scope the word search.
219     *
220     * @param qtext A search phrase.
221     *
222     * @return The current query definition for use in continued building.
223     */
224    CombinedQueryDefinition term(String qtext);
225
226    /**
227     * Add SPARQL queries to constraint results to only documents that have the matched triples.
228     *
229     * @param sparql The SPARQL query.
230     *
231     * @return The current query definition for use in continued building.
232     */
233    CombinedQueryDefinition sparql(String sparql);
234
235    /**
236     * Specified the entity type to use when building the query.  There are various annotations that can be used on your
237     * entity classes.  If you want those to be used in making decisions on how to build certain options then make sure
238     * to use this to "add" it to the query.  If not specified then the defaults will always be used.
239     *
240     * @param entityClass The type of the entity.
241     *
242     * @return The current query definition for use in continued building.
243     */
244    CombinedQueryDefinition type(Class entityClass);
245
246    /**
247     * Override the default mapping context to use when building queries.
248     *
249     * @param mappingContext A mapping context that maps entity information for methods that specify entity types.
250     *
251     * @return The current query definition for use in continued building.
252     */
253    CombinedQueryDefinition context(MappingContext mappingContext);
254
255    /**
256     * Indicates whether or not this combined query is a raw Query By Example query.
257     *
258     * @return True if the combined query contains a raw QBE.
259     */
260    boolean isQbe();
261
262    /**
263     * Indicates whether or not a limit has been specified on this query.
264     *
265     * @return True if the query is limiting.
266     */
267    boolean isLimiting();
268
269    /**
270     * Gets the configured limit for the current limit.
271     *
272     * @return The limit, or -1 if not set.
273     */
274    int getLimit();
275
276    /**
277     * Used by the MarkLogic Java Client Library to get the "low-level" representation of the query that can be submitted
278     * to the MarkLogic REST API.
279     *
280     * @return JSON/XML string of a combined query.
281     */
282    String serialize();
283}
