001package org.springframework.data.marklogic.repository.query;
002
003import com.marklogic.client.document.ServerTransform;
004import com.marklogic.client.query.StructuredQueryDefinition;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007import org.springframework.data.marklogic.core.MarkLogicOperations;
008import org.springframework.data.marklogic.core.convert.ServerTransformer;
009import org.springframework.data.marklogic.repository.Query;
010import org.springframework.data.marklogic.repository.query.MarkLogicQueryExecution.*;
011import org.springframework.data.repository.query.*;
012import org.springframework.util.Assert;
013import org.springframework.util.StringUtils;
014import sun.reflect.generics.tree.ReturnType;
015
016import java.util.Arrays;
017
018import static org.springframework.data.marklogic.repository.query.CombinedQueryDefinitionBuilder.combine;
019
020public abstract class AbstractMarkLogicQuery implements RepositoryQuery {
021
022    private static final Logger LOG = LoggerFactory.getLogger(AbstractMarkLogicQuery.class);
023    private final MarkLogicQueryMethod method;
024    private final MarkLogicOperations operations;
025
026    public AbstractMarkLogicQuery(MarkLogicQueryMethod method, MarkLogicOperations operations) {
027        Assert.notNull(operations, "MarkLogicOperations must not be null!");
028        Assert.notNull(method, "MarkLogicQueryMethod must not be null!");
029
030        this.method = method;
031        this.operations = operations;
032    }
033
034    @Override
035    public Object execute(Object[] values) {
036        // TODO: projections are not enabled until we can add extracts to limit properties returned (based on projection interface)
037        ParameterAccessor accessor = new ParametersParameterAccessor(method.getParameters(), values);
038        StructuredQueryDefinition query = createQuery(accessor);
039
040//        ResultProcessor processor = method.getResultProcessor().withDynamicProjection(accessor);
041        ResultProcessor processor = method.getResultProcessor();
042        ReturnedType returnedType = processor.getReturnedType();
043        Class typeToRead = returnedType.getDomainType();
044
045        // Add transforms and extracts to the query, if they are in the annotations
046        query = transform(query, typeToRead, accessor);
047        query = extracts(query);
048
049        if (LOG.isDebugEnabled()) {
050            LOG.debug("Executing query " + query.serialize());
051        }
052
053//        return processor.processResult(getExecution(accessor).execute(query, typeToRead));
054        return getExecution(accessor).execute(query, typeToRead);
055    }
056
057    @Override
058    public QueryMethod getQueryMethod() {
059        return method;
060    }
061
062    /**
063     * Add a server transform to the created query, if one is specified.
064     */
065    private StructuredQueryDefinition transform(StructuredQueryDefinition query, Class typeToRead, ParameterAccessor accessor) {
066        Query queryAnnotation = method.getQueryAnnotation();
067        if (queryAnnotation != null) {
068            if (queryAnnotation.transformer() != ServerTransformer.class) {
069                ServerTransform reader = operations.getQueryMapper()
070                        .getTransformer(queryAnnotation.transformer())
071                        .reader(operations.getConverter().getMappingContext().getPersistentEntity(typeToRead), accessor);
072                query.setResponseTransform(reader);
073            } else if (StringUtils.hasText(queryAnnotation.transform())) {
074                query.setResponseTransform(new ServerTransform(queryAnnotation.transform()));
075            }
076        }
077        return query;
078    }
079
080    /**
081     * Add extracts to the created query, if any are specified.
082     *
083     * @param query
084     * @return
085     */
086    private StructuredQueryDefinition extracts(StructuredQueryDefinition query) {
087        String[] extracts = method.getExtracts();
088        if (extracts != null && extracts.length > 0) {
089            query = combine(query).extracts(Arrays.asList(extracts));
090        }
091        return query;
092    }
093
094    private MarkLogicQueryExecution getExecution(ParameterAccessor accessor) {
095        if (isDeleteQuery()) {
096            return new DeleteExecution(operations);
097        } else if (isExistsQuery()) {
098            return new ExistsExecution(operations);
099        } else if (isCountQuery()) {
100            return new CountExecution(operations);
101        } else if (method.isStreamingQuery()) {
102            return new StreamingExecution(operations, accessor.getPageable());
103        } else if (method.isSliceQuery() || method.isPageQuery()) {
104            if (method.isFacetedQuery()) {
105                return new FacetedPageExecution(operations, accessor.getPageable());
106            } else {
107                return new PagedExecution(operations, accessor.getPageable());
108            }
109        } else if (method.isCollectionQuery()) {
110            return new EntityListExecution(operations);
111        } else {
112            return new SingleEntityExecution(operations);
113        }
114    }
115
116    protected abstract StructuredQueryDefinition createQuery(ParameterAccessor accessor);
117
118    protected abstract boolean isCountQuery();
119    protected abstract boolean isExistsQuery();
120    protected abstract boolean isDeleteQuery();
121}
