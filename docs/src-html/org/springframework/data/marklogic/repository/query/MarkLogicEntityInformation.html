001package org.springframework.data.marklogic.repository.query;
002
003import org.springframework.data.repository.core.EntityInformation;
004
005import java.io.Serializable;
006
007public interface MarkLogicEntityInformation <T, ID extends Serializable> extends EntityInformation<T, ID> {
008
009    /**
010     * Returns the name of the collection the entity shall be persisted to.
011     *
012     * @return
013     */
014    String getCollectionName();
015
016    /**
017     * Returns the attribute that the id will be persisted to.
018     *
019     * @return
020     */
021    String getIdAttribute();
022}
