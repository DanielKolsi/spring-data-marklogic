031/**
032 * Allow us to keep a running build of query/options so we can set them throughout the different processing levels.  The
033 * CombinedQueryDefinition in the driver library is for a specific purpose and not modifiable, nor is it passable down
034 * to the template methods.  Since the structure is well-defined as part of MarkLogic's REST API we can create the appropriate
035 * combined query from what we have.
036 */
037public class CombinedQueryDefinitionBuilder extends AbstractQueryDefinition implements CombinedQueryDefinition {

063    private CombinedQueryDefinitionBuilder(StructuredQueryDefinition query) {
064        this();
065        if (query instanceof CombinedQueryDefinitionBuilder) {
066            CombinedQueryDefinitionBuilder builder = (CombinedQueryDefinitionBuilder) query;
067            this.structuredQuery = builder.getStructuredQuery();
068            this.qbe = builder.getQbe();
069            this.extracts = builder.getExtracts();
070            this.qtext = builder.getQtext();
071            this.options = builder.getOptions();
072            this.qbeFormat = builder.getQbeFormat();
073            this.criteria = builder.getCriteria();
074            this.limit = builder.getLimit();
075            this.selected = builder.getSelected();
076            this.entityClass = builder.getEntityClass();
077            this.mappingContext = builder.getMappingContext();
078        } else {
079            this.structuredQuery = query;
080        }
081
082        if (query != null) {
083            collections(query.getCollections());
084            directory(query.getDirectory());
085            optionsName(query.getOptionsName());
086            transform(query.getResponseTransform());
087            setCriteria(query.getCriteria());
088        }
089    }

091    private CombinedQueryDefinitionBuilder() {
092        super();
093        this.mappingContext = new MarkLogicMappingContext();
094        this.qb = new StructuredQueryBuilder();
095        this.options = new ArrayList<>();
096        this.extracts = new ArrayList<>();
097
098        // To allow "javascript objects" for the query language
099        objectMapper
100                .configure(JsonParser.Feature.ALLOW_SINGLE_QUOTES, true)
101                .configure(JsonParser.Feature.ALLOW_UNQUOTED_FIELD_NAMES, true);
102
103    }

105    // Allows us to generate a "random" name for a JSON property so we can support multiple options with the same name
106    private String unique(String name) {
107        return name + "%" + Math.round(Math.random() * 1000);
108    }

110    @Override
111    public String serialize() {
112        List<String> optionsToSerialize = new ArrayList<>(options);
113
114        if (limit >= 0) optionsToSerialize.add(format("<page-length>%s</page-length>", limit));
115
116        if (extracts.size() > 0) {
117            StringBuilder extract = new StringBuilder();
118            extract.append(format("<extract-document-data selected='%s'>", selected));
119
120            extracts.forEach(path -> extract.append(format("<extract-path>%s</extract-path>", path)));
121
122            extract.append("</extract-document-data>");
123            optionsToSerialize.add(extract.toString());
124        }
125
126        if (isQbe() && qbeFormat == Format.JSON) {
127
128            ObjectNode search = factory.objectNode();
129
130            if (qbe != null) {
131                try {
132                    search.set("$query", objectMapper.readTree(qbe.toString()));
133                } catch (IOException ex) {
134                    throw new IllegalArgumentException(qbe.toString());
135                }
136            }
137
138            if (!optionsToSerialize.isEmpty()) {
139                ObjectNode optionsJson = factory.objectNode();
140                optionsToSerialize.stream()
141                        .map(XML::toJSONObject)
142                        .map(JSONObject::toString)
143                        .map(option -> {
144                            try {
145                                return objectMapper.readTree(option);
146                            } catch (IOException ex) {
147                                throw new IllegalArgumentException(option);
148                            }
149                        })
150                        .forEachOrdered(option -> {
151                            // Fix path indexes - they don't convert "nicely" like many of the other indexes
152                            String index = (String) option.at("/sort-order/path-index").asText();
153                            if (!StringUtils.isEmpty(index)) {
154                                optionsJson
155                                        .set(unique("sort-order"), factory.objectNode()
156                                                .put("direction", option.at("/sort-order/direction").asText())
157                                                .set("path-index", factory.objectNode()
158                                                        .put("text", index)
159                                                )
160                                        );
161                            } else {
162                                option.fieldNames().forEachRemaining(key -> optionsJson.set(unique(key), option.get(key)));
163                            }
164                        });
165
166                search.set("options", optionsJson);
167            } else {
168                // We need an empty options node or the REST API is unhappy, which makes us unhappy
169                search.set("options", factory.objectNode());
170            }
171
172            // Make sure it is "proper" JSON so that MarkLogic is happy
173            return factory.objectNode().set("search", search).toString()
174                    .replaceAll("%[0-9]*\":", "\":"); // get rid of the "unique" identifiers for properties
175
176        } else if (isQbe() || structuredQuery != null ||
177                !optionsToSerialize.isEmpty() ||
178                hasText(qtext) ||
179                hasText(sparql)) {
180            StringBuilder search = new StringBuilder();
181
182            search.append("<search xmlns=\"http://marklogic.com/appservices/search\">");
183
184            if (isQbe() && qbeFormat == Format.XML) {
185                // TODO: Can we put the structured query in "additional-query" if the "main" query is a QBE?
186                // TODO: Support "true" XML QBE instead of converting the JSON to XML?
187                search.append("<q:qbe xmlns:q=\"http://marklogic.com/appservices/querybyexample\">");
188                JSONObject json = new JSONObject(qbe.toString());
189                // TODO: Somehow support namespaces?
190                search.append(XML.toString(json, "q:query").replace("<q:query", "<q:query xmlns=\"\""));
191                search.append("</q:qbe>");
192            } else if (structuredQuery != null) {
193                search.append(structuredQuery.serialize());
194            }
195
196            if (!optionsToSerialize.isEmpty())
197                search.append("<options>")
198                        .append(optionsToSerialize.stream().collect(Collectors.joining()))
199                        .append("</options>");
200
201            if (!StringUtils.isEmpty(qtext))
202                search.append("<qtext>")
203                        .append(qtext)
204                        .append("</qtext>");
205
206            if (!StringUtils.isEmpty(sparql))
207                search.append("<sparql>")
208                        .append(qtext)
209                        .append("</sparql>");
210
211            search.append("</search>");
212            return search.toString();
213        }
214        return "";
215    }

270    /**
271     * Create an array of queries, starting with the current structured queries and adding the specified ones.
272     *
273     * @param queries Variable number of structured query definitions.
274     *
275     * @return An array of structured query definitions that can be sent to and() and or(), or whatever takes varargs.
276     */
277    private StructuredQueryDefinition[] collect(StructuredQueryDefinition... queries) {
278        Set<StructuredQueryDefinition> current = new HashSet<>(singletonList(structuredQuery));
279        Collections.addAll(current, queries);
280        return current.toArray(new StructuredQueryDefinition[0]);
281    }

285        // Preserve any current collections set on the query so this can be called multiple times without destroying anything
286        Set<String> current = new HashSet<>(asList(getCollections()));
287        Collections.addAll(current, collections);
288        setCollections(current.toArray(new String[0]));
289        return this;

311    @Override
312    public CombinedQueryDefinition sort(Sort sort) {
313        if (sort != null && sort.iterator().hasNext()) {
314            final MarkLogicPersistentEntity entity = entityClass != null ? mappingContext.getPersistentEntity(entityClass) : null;
315
316            sort.forEach(order -> {
317                String propertyName = order.getProperty();
318                String direction = asMLSort(order.getDirection());
319
320                String path = null;
321                // If there is an entity then we can determine the configuration of the index from it, otherwise we just
322                // default to a path index. An error will be thrown by the database if the index is not created, though.
323                if (entity != null) {
324                    MarkLogicPersistentProperty property = (MarkLogicPersistentProperty) entity.getPersistentProperty(propertyName);
325                    if (property != null) {
326                        // If the user specified type of "PATH" (default), or a path was specified, do we set it.  If
327                        // they specify type of "ELEMENT" then we don't set a path and let it fall through into the
328                        // logic of creating an element sort
329                        if (property.getIndexType() == IndexType.PATH && !StringUtils.isEmpty(property.getPath())) {
330                            path = property.getPath();
331                        }
332                    } else {
333                        // If the property was not found (user probably specified a path) then we default to path index
334                        path = "/"+ order.getProperty();
335                    }
336                } else {
337                    path = "/"+ order.getProperty();
338                }
339
340                // If any of the conditions above made it seem like this needs to be an element sort, then do so,
341                // otherwise default to a path index sort
342                if (!hasText(path)) {
343                    sort(propertyName, direction, IndexType.ELEMENT);
344                } else {
345                    sort(path, direction, IndexType.PATH);
346                }
347            });
348        }
349
350        return this;
351    }

353    @Override
354    public CombinedQueryDefinition sort(Sort sort, IndexType type) {
355        if (sort != null) {
356            sort.forEach(order -> sort(order.getProperty(), asMLSort(order.getDirection()), type));
357        }
358        return this;
359    }

361    @Override
362    public CombinedQueryDefinition sort(String propertyName, String order, IndexType type) {
363        if (hasText(propertyName) && hasText(order) && type != null) {
364            if (type == IndexType.PATH) {
365                options(format("" +
366                        "<sort-order direction='%s'>" +
367                        "   <path-index>%s</path-index>" +
368                        "</sort-order>", order, propertyName));
369            } else {
370                options(format("" +
371                        "<sort-order direction='%s'>" +
372                        "   <element ns='' name='%s'/>" +
373                        "</sort-order>", order, propertyName));
374            }
375        }
376
377        return this;
378    }
379
380    private String asMLSort(Sort.Direction direction) {
381        if (Sort.Direction.DESC.equals(direction)) {
382            return "descending";
383        } else {
384            return "ascending";
385        }
386    }

437
438        if (entityClass != null) {
439            MarkLogicPersistentEntity entity = mappingContext.getPersistentEntity(entityClass);

474    private CombinedQueryDefinition and(StructuredQueryDefinition query) {
475        if (structuredQuery != null)
476            structuredQuery = qb.and(structuredQuery, query);
477        else
478            structuredQuery = query;
479        return this;
