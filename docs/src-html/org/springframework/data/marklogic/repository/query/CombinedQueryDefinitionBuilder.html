030/**
031 * Allow us to keep a running build of query/options so we can set them throughout the different processing levels.  The
032 * CombinedQueryDefinition in the driver library is for a specific purpose and not modifiable, nor is it passable down
033 * to the template methods.  Since the structure is well-defined as part of MarkLogic's REST API we can create the appropriate
034 * combined query from what we have.
035 */
036public class CombinedQueryDefinitionBuilder extends AbstractQueryDefinition implements CombinedQueryDefinition {

062    private CombinedQueryDefinitionBuilder(StructuredQueryDefinition query) {
063        this();
064        if (query instanceof CombinedQueryDefinitionBuilder) {
065            CombinedQueryDefinitionBuilder builder = (CombinedQueryDefinitionBuilder) query;
066            this.structuredQuery = builder.getStructuredQuery();
067            this.qbe = builder.getQbe();
068            this.extracts = builder.getExtracts();
069            this.qtext = builder.getQtext();
070            this.options = builder.getOptions();
071            this.qbeFormat = builder.getQbeFormat();
072            this.criteria = builder.getCriteria();
073            this.limit = builder.getLimit();
074            this.selected = builder.getSelected();
075            this.entityClass = builder.getEntityClass();
076            this.mappingContext = builder.getMappingContext();
077        } else {
078            this.structuredQuery = query;
079        }
080
081        if (query != null) {
082            collections(query.getCollections());
083            directory(query.getDirectory());
084            optionsName(query.getOptionsName());
085            transform(query.getResponseTransform());
086            setCriteria(query.getCriteria());
087        }
088    }
089
090    private CombinedQueryDefinitionBuilder() {
091        super();
092        this.mappingContext = new MarkLogicMappingContext();
093        this.qb = new StructuredQueryBuilder();
094        this.options = new ArrayList<>();
095        this.extracts = new ArrayList<>();
096
097        // To allow "javascript objects" for the query language
098        objectMapper
099                .configure(JsonParser.Feature.ALLOW_SINGLE_QUOTES, true)
100                .configure(JsonParser.Feature.ALLOW_UNQUOTED_FIELD_NAMES, true);
101
102    }

104    // Allows us to generate a "random" name for a JSON property so we can support multiple options with the same name
105    private String unique(String name) {
106        return name + "%" + Math.round(Math.random() * 1000);
107    }

109    @Override
110    public String serialize() {
111        List<String> optionsToSerialize = options;
112
113        if (limit >= 0) optionsToSerialize.add(format("<page-length>%s</page-length>", limit));
114
115        if (extracts.size() > 0) {
116            StringBuilder extract = new StringBuilder();
117            extract.append(format("<extract-document-data selected='%s'>", selected));
118
119            extracts.forEach(path -> extract.append(format("<extract-path>%s</extract-path>", path)));
120
121            extract.append("</extract-document-data>");
122            optionsToSerialize.add(extract.toString());
123        }
124
125        if (isQbe() && qbeFormat == Format.JSON) {
126
127            ObjectNode search = factory.objectNode();
128
129            if (qbe != null) {
130                try {
131                    search.set("$query", objectMapper.readTree(qbe.toString()));
132                } catch (IOException ex) {
133                    throw new IllegalArgumentException(qbe.toString());
134                }
135            }
136
137            if (!optionsToSerialize.isEmpty()) {
138                ObjectNode optionsJson = factory.objectNode();
139                optionsToSerialize.stream()
140                        .map(XML::toJSONObject)
141                        .map(JSONObject::toString)
142                        .map(option -> {
143                            try {
144                                return objectMapper.readTree(option);
145                            } catch (IOException ex) {
146                                throw new IllegalArgumentException(option);
147                            }
148                        })
149                        .forEachOrdered(option -> {
150                            // Fix path indexes - they don't convert "nicely" like many of the other indexes
151                            String index = (String) option.at("/sort-order/path-index").asText();
152                            if (!StringUtils.isEmpty(index)) {
153                                optionsJson
154                                        .set(unique("sort-order"), factory.objectNode()
155                                                .put("direction", option.at("/sort-order/direction").asText())
156                                                .set("path-index", factory.objectNode()
157                                                        .put("text", index)
158                                                )
159                                        );
160                            } else {
161                                option.fieldNames().forEachRemaining(key -> optionsJson.set(unique(key), option.get(key)));
162                            }
163                        });
164
165                search.set("options", optionsJson);
166            } else {
167                // We need an empty options node or the REST API is unhappy, which makes us unhappy
168                search.set("options", factory.objectNode());
169            }
170
171            // Make sure it is "proper" JSON so that MarkLogic is happy
172            return factory.objectNode().set("search", search).toString()
173                    .replaceAll("%[0-9]*\":", "\":"); // get rid of the "unique" identifiers for properties
174
175        } else if (isQbe() || structuredQuery != null ||
176                !optionsToSerialize.isEmpty() ||
177                hasText(qtext) ||
178                hasText(sparql)) {
179            StringBuilder search = new StringBuilder();
180
181            search.append("<search xmlns=\"http://marklogic.com/appservices/search\">");
182
183            if (isQbe() && qbeFormat == Format.XML) {
184                // TODO: Can we put the structured query in "additional-query" if the "main" query is a QBE?
185                // TODO: Support "true" XML QBE instead of converting the JSON to XML?
186                search.append("<q:qbe xmlns:q=\"http://marklogic.com/appservices/querybyexample\">");
187                JSONObject json = new JSONObject(qbe.toString());
188                // TODO: Somehow support namespaces?
189                search.append(XML.toString(json, "q:query").replace("<q:query", "<q:query xmlns=\"\""));
190                search.append("</q:qbe>");
191            } else if (structuredQuery != null) {
192                search.append(structuredQuery.serialize());
193            }
194
195            if (!optionsToSerialize.isEmpty())
196                search.append("<options>")
197                        .append(optionsToSerialize.stream().collect(Collectors.joining()))
198                        .append("</options>");
199
200            if (!StringUtils.isEmpty(qtext))
201                search.append("<qtext>")
202                        .append(qtext)
203                        .append("</qtext>");
204
205            if (!StringUtils.isEmpty(sparql))
206                search.append("<sparql>")
207                        .append(qtext)
208                        .append("</sparql>");
209
210            search.append("</search>");
211            return search.toString();
212        }
213        return "";
214    }

221    @Override
222    public RawQueryByExampleDefinition getRawQbe() {
223        if (qbe != null) {
224            // Copy all the configuration from the combined query object so we don't loose any information as we pass back just the raw query object
225            qbe.setCollections(getCollections());
226            qbe.setDirectory(getDirectory());
227            qbe.setResponseTransform(getResponseTransform());
228            qbe.setOptionsName(getOptionsName());
229            qbe.setHandle(new StringHandle(serialize()).withFormat(qbeFormat));
230        }
231        return qbe;
232    }

259        // Preserve any current collections set on the query so this can be called multiple times without destroying anything

285    @Override
286    public CombinedQueryDefinition sort(Sort sort) {
287        if (sort != null && sort.iterator().hasNext()) {
288            final MarkLogicPersistentEntity entity = entityClass != null ? mappingContext.getPersistentEntity(entityClass) : null;
289
290            sort.forEach(order -> {
291                String propertyName = order.getProperty();
292                String direction = asMLSort(order.getDirection());
293
294                String path = null;
295                // If there is an entity then we can determine the configuration of the index from it, otherwise we just
296                // default to a path index. An error will be thrown by the database if the index is not created, though.
297                if (entity != null) {
298                    MarkLogicPersistentProperty property = (MarkLogicPersistentProperty) entity.getPersistentProperty(propertyName);
299                    if (!StringUtils.isEmpty(property.getPath())) {
300                        path = property.getPath();
301                    }
302                } else {
303                    path = "/"+ order.getProperty();
304                }
305
306                if (hasText(path)) {
307                    sort(path, direction, IndexType.PATH);
308                } else {
309                    sort(propertyName, direction, IndexType.ELEMENT);
310                }
311            });
312        }
313
314        return this;
315    }
316
317    @Override
318    public CombinedQueryDefinition sort(Sort sort, IndexType type) {
319        if (sort != null) {
320            sort.forEach(order -> sort(order.getProperty(), asMLSort(order.getDirection()), type));
321        }
322        return this;
323    }

325    @Override
326    public CombinedQueryDefinition sort(String propertyName, String order, IndexType type) {
327        if (hasText(propertyName) && hasText(order) && type != null) {
328            if (type == IndexType.PATH) {
329                options(format("" +
330                        "<sort-order direction='%s'>" +
331                        "   <path-index>%s</path-index>" +
332                        "</sort-order>", order, propertyName));
333            } else {
334                options(format("" +
335                        "<sort-order direction='%s'>" +
336                        "   <element ns='' name='%s'/>" +
337                        "</sort-order>", order, propertyName));
338            }
339        }
340
341        return this;
342    }
343
344    private String asMLSort(Sort.Direction direction) {
345        if (Sort.Direction.DESC.equals(direction)) {
346            return "descending";
347        } else {
348            return "ascending";
349        }
350    }

398    @Override
399    public CombinedQueryDefinition type(Class entityClass) {
400        this.entityClass = entityClass;
401
402        if (entityClass != null) {
403            MarkLogicPersistentEntity entity = mappingContext.getPersistentEntity(entityClass);
404
405            if (entity != null && entity.getTypePersistenceStrategy() == TypePersistenceStrategy.COLLECTION) {
406                collections(entity.getTypeName());
407            }
408        }
409
410        return this;
411    }

438    private CombinedQueryDefinition and(StructuredQueryDefinition query) {
439        if (structuredQuery != null)
440            structuredQuery = qb.and(structuredQuery, query);
441        else
442            structuredQuery = query;
443        return this;
