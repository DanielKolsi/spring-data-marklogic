001package org.springframework.data.marklogic.repository;
002
003import com.marklogic.client.io.Format;
004import org.springframework.core.annotation.AliasFor;
005import org.springframework.data.annotation.QueryAnnotation;
006import org.springframework.data.marklogic.repository.query.QueryType;
007import org.springframework.data.marklogic.repository.query.SelectedMode;
008
009import java.lang.annotation.*;
010
011/**
012 * Declare QBE queries directly on repository methods. This allows cleaner queries for more complex cases. This also
013 * allows the "tweaking" of basic search parameters in a repository interface without having to use MarkLogicTemplate
014 * directly to set them.
015 *
016 */
017@Retention(RetentionPolicy.RUNTIME)
018@Target({ ElementType.METHOD, ElementType.ANNOTATION_TYPE })
019@Documented
020@QueryAnnotation
021public @interface Query {
022
023    /**
024     * Takes a MarkLogic QBE JSON string to define the actual query to be executed. This one will take precedence over the
025     * method name then.
026     *
027     * @return
028     */
029    @AliasFor("query")
030    String value() default "";
031
032    @AliasFor("value")
033    String query() default "";
034
035    /**
036     * Sometimes it is better to force the use of a range index for equality checks (to point to specific properties instead of
037     * any property in a hierarchy).  This allows the ability to do so.  Changing to the RANGE type would require a range index
038     * created, and possible the @Indexed attribute with the path set on the property in the POJO class.
039     *
040     * @return
041     */
042    QueryType type() default QueryType.VALUE;
043
044    /**
045     * The document format to match, either XML or JSON.  Currently can't match a mix using this approach, you would need
046     * a StructuredQueryDefinition using the query builder in order to match both formats.  Using the MarkLogic
047     * {@link com.marklogic.client.io.Format} instead of {@link org.springframework.data.marklogic.core.mapping.DocumentFormat}
048     * since we allow "UNKNOWN" as the default.
049     *
050     * Setting this to "UNKNOWN", BINARY" OR "TEXT" would just end up setting the queries as JSON format, since that is
051     * default query "language"
052     *
053     * @return
054     */
055    Format format() default Format.UNKNOWN;
056
057    /**
058     * Defines the properties that should be returned for the given query. Note that only these properties will make it into the
059     * domain object returned.
060     *
061     * @return
062     */
063    String[] extract() default {};
064
065    /**
066     * Used in conjunction with the extract paths.  This determines how the extracted nodes are returned.  By default this
067     * returns specified nodes in their original hierarchy, but you can also specify to just return the extracted nodes, or to
068     * exclude the specified nodes.
069     *
070     * @return
071     */
072    SelectedMode selected() default SelectedMode.HIERARCHICAL;
073
074    /**
075     * To specify any query options to use in the query.  Since the query could end up as a range query or a text query of some
076     * kind, or MarkLogic could add additional options, this is not limited by an enumeration but the query will fail with
077     * incorrect options.  These options are applied to all the value/range/text queries that are created, so if fine-
078     * tuning is required at that level a custom Repository is required.
079     *
080     * @return
081     */
082    String[] options() default {};
083
084    /**
085     * Instead of specifying limited query options individually you can persist an options configuration to the database
086     * and just reference it to be used in the query.  This allows full configuration of the query for the annotated query.
087     *
088     * @return
089     */
090    String optionsName() default "";
091
092    /**
093     * The name of a transform to use when returning/saving documents (depends on the type of operations that is annotated).
094     * This transform must have been previously configured through the REST API otherwise the operation will fail.
095     *
096     * @return
097     */
098    String transform() default "";
099}
