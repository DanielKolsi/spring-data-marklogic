001package org.springframework.data.marklogic.domain.facets;
002
003import com.marklogic.client.query.FacetResult;
004import org.springframework.data.domain.PageImpl;
005import org.springframework.data.domain.Pageable;
006
007import java.util.Arrays;
008import java.util.List;
009import java.util.stream.Collectors;
010
011/**
012 * Extension of Spring's {@link org.springframework.data.domain.Page} that adds MarkLogic's facet results to a search
013 * result page.
014 *
015 * @param <T>
016 */
017public class FacetedPage<T> extends PageImpl<T> {
018
019    private List<FacetResultDto> facets;
020
021    public FacetedPage(List<T> content, Pageable pageable, long total, FacetResult[] facets) {
022        super(content, pageable, total);
023        this.facets = Arrays.stream(facets)
024                .map(FacetResultDto::new)
025                .collect(Collectors.toList());
026    }
027
028    public List<FacetResultDto> getFacets() {
029        return facets;
030    }
031}
